120 public void Field(String[] params) throws Exception {
121 try{
122 if (game == null) throw new Exception();
123
124 HashMap<String, String> options = new HashMap<>();
125 String line;
126 while (!(line = sc.nextLine()).equals("end")){
127 String[] command = line.split(" ");
128 options.put(command[0], command[1]);
129 }
130
131 Field f = null;
132 String arg = options.get("Param");
133
134 if (arg == null){ //default ctor
135 f = (Field) createObject("model.map." + options.get("Type"));
136 } else{
137 //csúnya, de ez van
138 switch(options.get("Type")){
139 case "Laboratory":
140 GeneticCode c1 = (GeneticCode) createObject("model.codes." + arg);
141 f = new Laboratory(game.AddGeneticCode(c1));
142 break;
143 case "InfectedLaboratory":
144 GeneticCode c2 = (GeneticCode) createObject("model.codes." + arg);
145 f = new InfectedLaboratory(game.AddGeneticCode(c2));
146 break;
147 case "Shelter":
148 Equipment e = (Equipment) createObject("model.equipments."+arg);
149 f = new Shelter(e);
150 break;
151 default:
152 break;
153 }
154 }
155 String eqType = options.get("Equipment");
156 if (eqType != null){
157 f.Drop((Equipment) createObject("model.equipments." + eqType));
158 }
159 if (options.get("Name") == null) throw new Exception(); //Name is mandatory!
160 if (fields.get(options.get("Name")) != null) throw new Exception(); //Field with Name already Exists!
161 f.setName(options.get("Name"));
162 fields.put(options.get("Name"), f);
163 game.AddField(f);
164 } catch (Exception e){
165 throw new Exception("Error in Field command format!");
166 }

196 @ProtoInput(name="Virologist")
197 public void Virologist(String[] params) throws Exception {
198 try{
199 if (game == null) throw new Exception();
200 String line;
201 Virologist v = new Virologist();
202 String startingPos = "";
203 while (!(line = sc.nextLine()).equals("end")){
204 String[] command = line.split(" ");
205 switch(command[0]){
206 case "Name":
207 v.setName(command[1]);
208 break;
209 case "Equipment":
210 v.AddEquipment((Equipment) createObject("model.equipments." + command[1]));
211 break;
212 case "Amino":
213 v.AddAminoAcid(Integer.parseInt(command[1]));
214 break;
215 case "Nucleo":
216 v.AddNucleotide(Integer.parseInt(command[1]));
217 break;
218 case "Agent":
219 Class<?> c= Class.forName("model.agents." + command[1]);
220 Agent a = (Agent) c.getConstructor(Integer.TYPE).newInstance(Integer.parseInt(command[2]));
221 a.Apply(v);
222 v.AddAgent(a);
223 a.ApplyStrategy(v);
224 break;
225 case "StartingPos":
226 startingPos = command[1]; //in case there are command after this that throw error we don't want to mess up the existing setup
227 break;
228 case "GeneticCode":
229 v.AddGeneticCode((GeneticCode) createObject("model.codes." + command[1]));
230 break;
231 case "ActionCount":
232 v.SetActionCount(Integer.parseInt(command[1]));
233 break;
234 default:
235 throw new Exception();
236 }
237 }
238 fields.get(startingPos).AddVirologist(v);
239 game.AddVirologist(v);
240 } catch (Exception e){
241 throw new Exception("Error in Virologist command format!");
242 }
